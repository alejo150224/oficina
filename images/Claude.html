    Para que la aplicación solicite ingresar nuevamente las credenciales después de un cierto tiempo, puedes utilizar la función en JavaScript. Esta función ejecuta una función o un fragmento de código después de un retraso especificado.setTimeout

    const loginForm = document.getElementById('login-form');
const usernameInput = document.getElementById('username');
const passwordInput = document.getElementById('password');

// Credenciales válidas de ejemplo
const validCredentials = {
  username: 'usuario',
  password: 'contraseña'
};

let loggedIn = false; // Variable para rastrear el estado de inicio de sesión
const sessionTimeout = 300000; // 5 minutos en milisegundos (300000 ms)

// Función para iniciar la sesión
function startSession() {
  loggedIn = true;
  window.location.href = 'todo-list.html';

  // Reiniciar el temporizador de sesión
  clearTimeout(sessionTimeoutId);
  sessionTimeoutId = setTimeout(endSession, sessionTimeout);
}

// Función para finalizar la sesión
function endSession() {
  loggedIn = false;
  window.location.href = 'login.html';
}

let sessionTimeoutId; // Variable para almacenar el ID del temporizador de sesión

loginForm.addEventListener('submit', (event) => {
  event.preventDefault();

  const username = usernameInput.value;
  const password = passwordInput.value;

  if (username === validCredentials.username && password === validCredentials.password) {
    // Credenciales válidas, iniciar la sesión
    startSession();
  } else {
    // Credenciales inválidas, mostrar un mensaje de error
    alert('Usuario o contraseña incorrectos');

// Verificar si el usuario ya ha iniciado sesión

  // Reiniciar el temporizador de sesión

    login.js:

    javascript
    const loginForm = document.getElementById('login-form');
const usernameInput = document.getElementById('username');
const passwordInput = document.getElementById('password');

// Credenciales válidas de ejemplo
const validCredentials = {
  username: 'usuario',
  password: 'contraseña'
};

let loggedIn = false; // Variable para rastrear el estado de inicio de sesión
const inactivityTimeout = 300000; // 5 minutos en milisegundos (300000 ms)

// Función para iniciar la sesión
function startSession() {
  loggedIn = true;
  window.location.href = 'todo-list.html';

  // Reiniciar el temporizador de inactividad
  clearTimeout(inactivityTimeoutId);
  resetInactivityTimer();
}

// Función para finalizar la sesión
function endSession() {
  loggedIn = false;
  window.location.href = 'login.html';
}

let inactivityTimeoutId; // Variable para almacenar el ID del temporizador de inactividad

// Función para reiniciar el temporizador de inactividad
function resetInactivityTimer() {
  clearTimeout(inactivityTimeoutId);
  inactivityTimeoutId = setTimeout(endSession, inactivityTimeout);
}

// Detectar actividad del usuario
document.addEventListener('mousemove', resetInactivityTimer);
document.addEventListener('keydown', resetInactivityTimer);

loginForm.addEventListener('submit', (event) => {
  event.preventDefault();

  const username = usernameInput.value;
  const password = passwordInput.value;

  if (username === validCredentials.username && password === validCredentials.password) {
    // Credenciales válidas, iniciar la sesión
    startSession();
  } else {
    // Credenciales inválidas, mostrar un mensaje de error
    alert('Usuario o contraseña incorrectos');
  }
});

// Verificar si el usuario ya ha iniciado sesión
if (loggedIn) {
  // Reiniciar el temporizador de inactividad
  resetInactivityTimer();
}
